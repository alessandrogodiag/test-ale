    <h1>Invia Messaggio WhatsApp</h1>
    <input type="text" id="nome" placeholder="Nome">
    <input type="tel" id="telefono" placeholder="Numero di telefono">
    <textarea id="messaggio" rows="4" placeholder="Messaggio personalizzato"></textarea>
    <button onclick="inviaMessaggio()">Invia Messaggio</button>

    <script>
        function inviaMessaggio() {
            const nome = document.getElementById('nome').value;
            const telefono = document.getElementById('telefono').value;
            let messaggio = document.getElementById('messaggio').value;

            // Sostituisce [NOME] nel messaggio con il nome inserito
            messaggio = messaggio.replace('[NOME]', nome);

            // Formatta il numero di telefono (rimuove spazi e caratteri non numerici)
            const telefonoFormattato = telefono.replace(/\D/g, '');

            // Codifica il messaggio per l'URL
            const messaggioCodificato = encodeURIComponent(messaggio);

            // Crea l'URL di WhatsApp
            const whatsappUrl = `https://wa.me/${telefonoFormattato}?text=${messaggioCodificato}`;

            // Apre WhatsApp in una nuova finestra
            window.open(whatsappUrl, '_blank');
        }
    </script>
